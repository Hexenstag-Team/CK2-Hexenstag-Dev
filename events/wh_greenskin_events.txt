#decade pulse event
character_event = {
    id = wh_greenskins.10
    hide_window = yes
    is_triggered_only = yes

    religion_group = greenskin_gods_group

    trigger = {
        prestige = 100
        OR = {
            trait = creature_orc
            trait = creature_goblin
        }
    }

    #run logic for growth, then grow ai greenskins and send notification to growing players
    immediate = {
        trigger_switch = {
            on_trigger = trait
            goblin_small = {
                if = {
                    limit = {
                        #a boss in the next size up bash greenskins so they don't grow (bigger bosses don't feel threatened)
                        NOT = {
                            liege = {
                                trait = goblin_average
                            }
                        }
                    }
                    if = {
                        limit = {
                            ai = yes
                        }
                        add_trait = goblin_average
                    }
                    else = {
                        character_event = {
                            id = wh_greenskins.11
                        }
                    }
                }
            }
            goblin_average = {
                if = {
                    limit = {
                        NOT = {
                            liege = {
                                trait = goblin_large
                            }
                        }
                        prestige = 250
                        OR = {
                            martial = 12
                            has_focus = focus_war
                        }
                    }
                    if = {
                        limit = {
                            ai = yes
                        }
                        add_trait = goblin_large
                    }
                    else = {
                        character_event = {
                            id = wh_greenskins.11
                        }
                    }
                }
            }
            goblin_large = {
                if = {
                    limit = {
                        NOT = {
                            liege = {
                                trait = goblin_huge
                            }
                        }
                        prestige = 500
                        martial = 12
                        OR = {
                            martial = 18
                            has_focus = focus_war
                        }
                    }
                    if = {
                        limit = {
                            ai = yes
                        }
                        add_trait = goblin_huge
                    }
                    else = {
                        character_event = {
                            id = wh_greenskins.11
                        }
                    }
                }
            }
            #this is a "win more" mechanic but that is how greenskins work
            goblin_huge = {
                if = {
                    limit = {
                        independent = yes
                        has_focus = focus_war
                        prestige = 2500
                    }
                    if = {
                        limit = {
                            ai = yes
                        }
                        change_martial = 1
                    }
                    else = {
                        character_event = {
                            id = wh_greenskins.11
                        }
                    }
                }
            }
            orc_small = {
                if = {
                    limit = {
                        NOT = {
                            liege = {
                                trait = orc_average
                            }
                        }
                    }
                    if = {
                        limit = {
                            ai = yes
                        }
                        add_trait = orc_average
                    }
                    else = {
                        character_event = {
                            id = wh_greenskins.11
                        }
                    }
                }
            }
            orc_average = {
                if = {
                    limit = {
                        NOT = {
                            liege = {
                                trait = orc_large
                            }
                        }
                        prestige = 250
                        OR = {
                            martial = 12
                            has_focus = focus_war
                        }
                    }

            }
            orc_large = {

    id = wh_greenskins.100
    hide_window = yes
    is_triggered_only = yes

    only_playable = yes

    trigger = {
        has_landed_title = k_cult_of_the_lady

        any_province = {

                    de_jure_liege_or_above = d_ostlichschattenwald

                #testing dynamic exemption for Chaos Dwarfs - if they have their satrapies working, the greenskins aren't a problem
                NOR = {
                    culture = dawi_zharr
                    culture = servile_hobgoblin
                }
            }
            random_list = {
                10 = {
                    trigger = {
                        region = world_badlands
                    }

                        id = wh_greenskins.101

                        random = 5
                    }
                    province_event = {
                        id = wh_greenskins.101
                        days = 105
                        random = 105
                    }
                }
                10 = {
                    trigger = {
                        OR = {

                        id = wh_greenskins.101

province_event = {
    id = wh_greenskins.101
    hide_window = yes
    is_triggered_only = yes

    immediate = {
        #regular "population control" effects go here
        if = {
            limit = {
                NOT = {
                    religion_group = greenskin_gods_group
                }
                has_wonder = yes
            }
            change_variable = {
                which = wild_greenskin_power
                value = -2
            }
        }
        if = {
            limit = {
                OR = {
                    fort_has_building = ft_greenskin_patrol
                    #ai doesn't build forts right now so it gets this for free if appropriate
                    NOR = {
                        religion_group = greenskin_gods_group
                        county = {
                            any_defacto_liege = {
                                ai = no
                            }
                        }
                    }
                }
            }
            change_variable = {
                which = wild_greenskin_power
                value = -2
            }
        }
        #if necessary, apply effects for strong wild greenskins
        if = {
            limit = {
                check_variable = {
                    which = wild_greenskin_power
                    value = 20
                }
            }
            if = {
                limit = {
                    owner = {
                        NAND = {
                            religion_group = greenskin_gods_group
                            #i.e. ai = no will always get revolts, though these are arguably an advantage
                            ai = yes
                        }
                    }
                }
                #revolt - needs randomness to spread these guys out because we want more of a cascade and less of a "bomb"
                random_list = {
                    10 = {
                        province_event = {
                            id = DR_greenskins.171
                            days = 15
                            random = 45
                        }
                    }
                    30 = {
                        province_event = {
                            id = DR_greenskins.171
                            days = 45
                            random = 75
                        }
                    }
                    30 = {
                        province_event = {
                            id = DR_greenskins.171
                            days = 75
                            random = 165
                        }
                    }
                    10 = {
                        province_event = {
                            id = DR_greenskins.171
                            days = 165
                            random = 315
                        }
                    }
                }
            }
            else = {
                #help for AI Greenskin rulers
                #if province is colonized, add a mob building
                if = {
                    limit = {
                        has_tribal = yes
                        OR = {
                            culture_group = goblin_group
                            culture_group = orc_group
                        }
                    }
                    province_event = {
                        id = wh_greenskins.110
                    }
                }
                #otherwise add troops
                else = {
                    owner = {
                        character_event = {
                            id = DR_greenskins.193
                        }
                    }
                    change_variable = {
                        which = wild_greenskin_power
                        value = -10
                    }
                }
            }
        }
        #otherwise increase wild_greenskin_power
        else = {
            random_list = {
                20 = {
                    #can't happen in greenskin homelands
                    trigger = {
                        NOT = {
                            region = world_badlands
                            region = world_darklands
                        }
                    }
                    #no change
                }
                30 = {
                    change_variable = {
                        which = wild_greenskin_power
                        value = 5
                    }
                }
                30 = {
                    change_variable = {
                        which = wild_greenskin_power
                        value = 8
                    }
                }
                20 = {
                    #can't happen in the fringe greenskin regions
                    trigger = {
                        NOR = {
                            region = world_chaos_wastes_eastern_steppes
                            region = world_old_world_border_princes
                            region = world_old_empire_solland
                            de_jure_liege_or_above = d_ostlichschattenwald
                        }
                    }
                    change_variable = {
                        which = wild_greenskin_power
                        value = 12
                    }
                }
            }
        }
        #finally, logic for adding and removing a warning modifier for the player if applicable
        if = {
            limit = {
                county = {
                    any_defacto_liege = {
                        ai = no
                        #no point in taunting nomad players
                        is_nomadic = no
                    }
                }
                check_variable = {
                    which = wild_greenskin_power
                    value = 10
                }
            }
            add_province_modifier = {
                which = wild_greenskin_danger_province
                duration = -1
            }
        }
        else_if = {
            limit = {
                NOT = {
                    check_variable = {
                        which = wild_greenskin_power
                        value = 10
                    }
                }
            }
            remove_province_modifier = wild_greenskin_danger_province
        }
    }
}
