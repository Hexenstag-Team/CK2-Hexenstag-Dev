	desc = EVTDESCA_DR_imperial_religion_unification

	picture = GFX_evt_icon

	is_triggered_only = yes

	option = {

			limit = {
				trait = patron_sigmar
			}
			set_high_god_name = GOD_SIGMAR
			set_global_flag = imperial_churches_unified_sigmar
			give_nickname = nick_the_pious
		}

		else_if = {
			limit = {
				trait = patron_taal_rhya
			}
			set_high_god_name = GOD_TAAL
			remove_god_names = yes
			add_god_name = {
				GOD_TAAL
				GOD_RHYA
				GOD_MANANN
				GOD_ULRIC
				GOD_MORR
				GOD_VERENA
				GOD_MYRMIDIA
				GOD_SHALLYA
				GOD_RANALD		
			}
			set_global_flag = imperial_churches_unified_taal_rhya
			give_nickname = nick_the_pious
			character_event = { id = swhunters.7 }
		}

		else_if = {
			limit = {
				trait = patron_ulric
			}
			set_high_god_name = GOD_ULRIC
			set_global_flag = imperial_churches_unified_ulric
			give_nickname = nick_the_pious
		}

		else_if = {
			limit = {
				NOR = {
					trait = patron_sigmar
					trait = patron_taal_rhya
					trait = patron_ulric
				}
			}
			set_global_flag = imperial_churches_unified
			give_nickname = nick_the_pious
		}

		hidden_tooltip = {
			religion_authority = { modifier = faith_triumphant }
		}
	}
}
