character_event = {
	id = wh_setup.11
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		OR = {
			trait = creature_slann
			AND = {
				religion = atruhayid
				is_feudal = yes
				capital_scope = {
					port = yes
				}
			}

		}
	}
	immediate = {
		if = {
			limit = {
				trait = creature_slann
			}
			set_special_character_title = MAGE_PRIEST
		}
		if = {
			limit = {
				religion = atruhayid
				is_feudal = yes
				capital_scope = {
					port = yes
				}
			}
			if = {
				limit = {
					tier = DUKE
				}
				if = {
					limit = { is_female = yes }
					set_special_character_title = CORSAIR_LADY
				}
				else = {
					limit = { is_female = no }
					set_special_character_title = CORSAIR_LORD
				}
				break = yes
			}
			else_if = {
				limit = {
					tier = KING
				}
				if = {
					limit = { is_female = yes }
					set_special_character_title = CORSAIR_PRINCESS
				}
				else = {
					limit = { is_female = no }
					set_special_character_title = CORSAIR_PRINCE
				}
				break = yes
			}
			else_if = {
				limit = {
					tier = EMPEROR
				}
				if = {
					limit = { is_female = yes }
					set_special_character_title = CORSAIR_QUEEN
				}
				else = {
					limit = { is_female = no }
					set_special_character_title = CORSAIR_KING
				}
				break = yes
			}
			else = {
				set_special_character_title = CORSAIR
			}
		}
	}
}
