character_event = {
	id = whBloodline.1
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		if = {
			limit = {
				NOT = {
					any_owned_bloodline = {
						has_bloodline_flag = burnt_by_flames_redeemed
						bloodline_is_active_for = PREV
					}
				}
			}
			if = {
				limit = {
					is_female = no
				}
				create_bloodline = { type = burnt_by_flames_male }
			}
			if = {
				limit = {
					is_female = yes
				}
				create_bloodline = { type = burnt_by_flames_female }
			}
		}
	}
}

character_event = {
	id = whBloodline.3

	hide_window = yes # Only fires for unplayable landless characters

	min_age = 45
	max_age = 300
	capable_only = yes
	prisoner = no

	trigger = {
		has_dlc = "The Old Gods"
		any_owned_bloodline = {
	    	has_bloodline_flag = burnt_by_flames
			bloodline_is_active_for = PREV
	    }
		top_liege = {
			religion = cadai
		}
		NOT = {
			has_game_rule = {
				name = adventurers
				value = none
			}
		}

		is_ruler = no
		NOT = { any_spouse = { is_ruler = yes } }

		diplomacy = 4
		martial = 6

		NOT = { trait = content }
		NOT = { trait = craven }


		NOT = { # Somewhat clunky way of checking that the character does not stand to inherit any titles
			any_heir_title = {
				always = yes
			}
		}

		NOT = {
			any_claim = {
				has_holder = yes
				holder_scope = {
					independent = yes
					OR = {
						is_merchant_republic = no
						NOT = {
							capital_holding = {
								de_jure_liege_or_above = PREVPREV
							}
						}
					}
				}
				OR = {
					tier = KING
					holder_scope = { tier = PREV }
				}
				ROOT = {
					NOR = {
						obedient = PREV
						non_interference = PREV
					}
				}
			}
		}
		NOT = { has_character_modifier = planning_duchy_adventure }
	}

	mean_time_to_happen = {
		days = 1
		modifier = {
			factor = 1.5
			trait = ambitious
		}
		modifier = {
			factor = 2
			has_character_flag = flag_denied_title
		}
		modifier = {
			factor = 2
			has_character_flag = demon_child_non_pagan
		}
		modifier = {
			factor = 2
			top_liege = {
				trait = burnt_by_asuryan
			}
		}
		#Don't annoy players by sending off family
		modifier = {
			factor = 0.2
			check_child_or_grandchild_of_active_player_character = yes
		}
		modifier = {
			factor = 1.5
			diplomacy = 12
		}
		modifier = {
			factor = 1.5
			martial = 12
		}
		modifier = {
			factor = 1.25
			trait = brave
		}
		modifier = {
			factor = 1.15
			trait = proud
		}
		modifier = {
			factor = 0.3
			trait = slothful
		}
		modifier = {
			factor = 0.3
			trait = humble
		}
		modifier = {
			factor = 0.3
			trait = kind
		}
	}


	immediate = {
		random_independent_ruler = {
			limit = {
				NOT = { culture_group = ROOT }
				is_landed = yes
				independent = yes
				NOT = { is_vassal_or_below = ROOT }
				any_realm_province = {
					OR = {
						AND = {
							port = yes
							ROOT = {
								NOT = { culture_group = kurgan_group }
								NOT = { culture_group = hung_group }
							}
							NOT = {
								distance = {
									who = ROOT
									value = 2000
								}
							}
						}
						any_neighbor_province = {
							owner = {
								same_realm = ROOT
							}
						}
					}
					NOR = {
						region = world_norsca
						region = world_chaos_wastes
						AND = {
							region = world_albion
							has_global_flag = mists_of_albion
						}
					}
				}
				ROOT = {
					NOR = {
						obedient = PREV
						non_interference = PREV
					}
				}
			}
			preferred_limit = {
				religion_group = greenskin_gods_group
			}
			preferred_limit = {
				religion = stromfels
			}
			preferred_limit = {
				religion = horned_rat
			}
			preferred_limit = {
				religion = norscan_gods
			}
			preferred_limit = {
				religion = old_ones
			}
			preferred_limit = {
				religion_group = chaos_gods_group
			}

			random_realm_province = {
				limit = {
					OR = {
						AND = {
							port = yes
							ROOT = {
								NOT = { culture_group = kurgan_group }
								NOT = { culture_group = hung_group }
							}
							NOT = {
								distance = {
									who = ROOT
									value = 2000
								}
							}
						}
						any_neighbor_province = {
							owner = {
								same_realm = ROOT
							}
						}
					}
					NOR = {
						region = world_norsca
						region = world_chaos_wastes
						AND = {
							region = world_albion
							has_global_flag = mists_of_albion
						}
					}
				}

				ROOT = {
					opinion = {
						who = PREVPREV
						years = 3
						modifier = opinion_duchy_adventure_target
					}
					add_character_modifier = {
						name = planning_duchy_adventure
						hidden = yes
						days = 730
					}
					character_event = {
						id = TOG.1212 # Fire the attack
						days = 730
					}
				}
				PREV = {
					character_event = {
						id = TOG.1211 # Warn the target
					}
				}
			}
		}
		end_inaccessibility_effect = yes #removes in hiding or in_seclusion

		set_character_flag = do_not_disturb
		character_event = { id = TOG.1215 days = 730 }
	}

	option = {
		name = OK
	}
}
