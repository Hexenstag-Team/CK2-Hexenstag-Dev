# Basic version without much content, but easily expanded with more results

#family treasure governing event
character_event = {
	id = wh_bloodline.1
	hide_window = yes
	is_triggered_only = yes

	only_playable = yes
	has_dlc = "Holy Fury"
	prisoner = no
	only_capable = yes

	trigger = {
		any_owned_bloodline = {
			has_bloodline_flag = bloodline_family_treasure
		}
		#shouldn't be possible for characters without these traits anyway
		OR = {
			trait = creature_human
			trait = creature_dwarf
		}
	}

	immediate = {
		random_list = {
			#trinket result
			20 = {
				character_event = {
					id = wh_bloodline.2
					days = 17
					random = 27
				}
			}
			#nice result
			20 = {
				character_event = {
					id = wh_bloodline.3
					days = 17
					random = 27
				}
			}
			#great result
			20 = {
				character_event = {
					id = wh_bloodline.4
					days = 17
					random = 27
				}
			}
			#better luck next time
			20 = {	}
		}
	}
}
