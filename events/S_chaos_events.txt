character_event = {
	id = schaos.1
	title = EVTTITLE_schaos1
	desc = EVTDESCschaos1
	picture = GFX_evt_chaos_hordes
	is_triggered_only = yes

	trigger = {

	}

	immediate = {
		set_variable = { which = popToAddCount value = 1000 }
		set_variable = { which = moneyToAddCount value = 0 }
		any_realm_province = {
			any_province_holding = {
				if = {
					limit = {
						holding_type = tribal
					}
					ROOT = {
						change_variable = { which = moneyToAddCount value = 10 }
					}
				}
				else = {
					ROOT = {
						change_variable = { which = moneyToAddCount value = 30 }
					}
				}
				ROOT = {
					change_variable = { which = popToAddCount value = 1000 }
				}
			}
		}
	}

	option = { name = EVTOPTAschaos1
		set_government_type = chaos_horde_government
		custom_tooltip = {
			text = convert_to_chaos_nomad
			any_realm_province = {
				any_province_holding = {
					convert_holdings_buildings_to_nomad_equiv = yes
				}
			}
			any_realm_province = {
				limit = {
					NOR = {
						county = {
							#The main city of the wastes
							title = c_zanbaijin
						}
						OR = {
							any_province_holding = {
								is_holy_site = chaos
							}
							any_province_holding = {
								is_holy_site = tzeentch
							}
							any_province_holding = {
								is_holy_site = nurgle
							}
							any_province_holding = {
								is_holy_site = slaanesh
							}
							any_province_holding = {
								is_holy_site = khorne
							}
							any_province_holding = {
								is_holy_site = norscan_gods
							}
						}
					}

				}
				population = 1000
				any_province_holding = {
					if = {
						limit = {
							NOT = { holding_type = nomad }
						}
						destroy_settlement = THIS
					}
				}
			}
		}
		population = popToAddCount
		wealth = moneyToAddCount
		hidden_tooltip = {
			capital_holding = {
				add_building = no_cattle_pens_1
				add_building = no_ox_wagon_1
				add_building = no_cattle_pens_2
				add_building = no_ox_wagon_2
			}
			set_variable = { which = popToAddCount value = 0 }
			set_variable = { which = moneyToAddCount value = 0 }
		}
		clr_character_flag = converted_to_chaos_horde
	}

	option = {
		name = Nevermind
		trigger = {
			ai = no
		}
		clr_character_flag = converted_to_chaos_horde
	}
}

	}

		#NOTE: it's important to run this check first so that we don't apply the more complicated logic below to so many characters
		if = {
			limit = {
				NOT = {
					num_of_friends = 1
				}
			}
			remove_character_modifier = khornate_pet
			remove_character_modifier = nurglite_pet
			remove_character_modifier = slaaneshi_pet
			remove_character_modifier = tzeentchian_pet
		}

				limit = {
					any_friend = { trait = khorne_demon }

					name = khornate_pet
