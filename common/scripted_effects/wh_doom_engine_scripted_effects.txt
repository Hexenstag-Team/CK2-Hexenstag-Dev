		limit = {
			check_variable = { which = "global_doom_counter" value = 3000 }
	   }

	}
}

	}

check_doom_not_negative = {
	if = {
		limit = {
			NOT = { check_variable = { which = "global_doom_counter" value = 1 } }
	   }
	   set_variable = { which = "global_doom_counter" value = 0 }
	}
}


add_doom_everchosen = {
	if = {
		limit = {
			any_realm_province = {
				NOT = { region = world_chaos_wastes }
			}
		}
		add_doom_progress_medium_effect = yes
	}
	else = {
		add_doom_progress_minor_effect = yes
	}
}

add_doom_prince = {
	if = {
		limit = {
			any_realm_province = {
				NOT = { region = world_chaos_wastes }
			}
		}
		random_list = {
			25 = {
				add_doom_progress_minor_effect = yes
				modifier = {
					factor = 0.3
					is_landed = no
				}
			}
			25 = {
				add_doom_progress_tiny_effect = yes
			}
			50 = { }
		}
	}
	else = {
		random_list = {
			5 = {
				add_doom_progress_minor_effect = yes
				modifier = {
					factor = 0.3
					is_landed = no
				}
			}
			45 = {
				add_doom_progress_tiny_effect = yes
			}
			50 = { }
		}
	}
}
