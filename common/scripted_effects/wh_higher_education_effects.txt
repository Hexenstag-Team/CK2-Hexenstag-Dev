find_close_university_effect = {
    if = {
        limit = {
            capital_scope = {
                wonder = {
                    has_wonder_upgrade = upgrade_university
                }
            }
        }
        capital_scope = {
            save_event_target_as = close_university_candidate
        }
    }
    else_if = {
        limit = {
            any_demesne_wonder_upgrade = upgrade_university
        }
        random_demesne_wonder_upgrade = {
            limit = {
                has_wonder_upgrade = upgrade_university
            }
            location = {
                save_event_target_as = close_university_candidate
            }
        }
    }
    else_if = {
        limit = {
            any_realm_wonder_upgrade = upgrade_university
        }
        random_realm_wonder_upgrade = {
            limit = {
                has_wonder_upgrade = upgrade_university
            }
            location = {
                save_event_target_as = close_university_candidate
            }
        }
    }
    else = {
        random_wonder_upgrade = {
            limit = {
                has_wonder_upgrade = upgrade_university
            }
            location = {
                save_event_target_as = close_university_candidate
            }
        }
    }
}

find_further_university_effect = {
    if = {
        limit = {
            any_demesne_wonder_upgrade = upgrade_university
            NOT = { province = event_target:close_university_candidate }
        }
        random_demesne_wonder_upgrade = {
            limit = {
                has_wonder_upgrade = upgrade_university
                NOT = { province = event_target:close_university_candidate }
            }
            location = {
                save_event_target_as = further_university_candidate
            }
        }
    }
    else_if = {
        limit = {
            any_realm_wonder_upgrade = upgrade_university
            NOT = { province = event_target:close_university_candidate }
        }
        random_realm_wonder_upgrade = {
            limit = {
                has_wonder_upgrade = upgrade_university
                NOT = { province = event_target:close_university_candidate }
            }
            location = {
                save_event_target_as = further_university_candidate
            }
        }
    }
    else = {
        random_wonder_upgrade = {
            limit = {
                has_wonder_upgrade = upgrade_university
                NOT = { province = event_target:close_university_candidate }
            }
            location = {
                save_event_target_as = further_university_candidate
            }
        }
    }
}

find_exotic_university_effect = {
    random_wonder_upgrade = {
        limit = {
            has_wonder_upgrade = upgrade_university
            NOR = {
                province = event_target:close_university_candidate
                province = event_target:further_university_candidate
            }
        }
        location = {
            save_event_target_as = exotic_university_candidate
        }
    }
}

degraded_by_further_education_effect = {
    random_list = {
        #no problems
        90 = {
            #all good
        }
        #mental strain
        5 = {
            modifier = {
                factor = 0.5
                trait = content
            }
            modifier = {
                factor = 2
                is_weak_trigger = yes
            }
            modifier = {
                factor = 2
                trait = stressed
            }
            modifier = {
                factor = 5
                trait = depressed
            }
            trigger_switch = {
                on_trigger = trait
                stressed = {
                    add_trait = depressed
                }
                depressed = {
                    random = {
                        chance = 10
                        death = {
                            death_reason = death_suicide
                        }
                    }
                }
                fallback = {
                    add_trait = stressed
                }
            }
        }
        #physical strain
        5 = {
            modifier = {
                factor = 2
                trait = diligent
            }
            modifier = {
                factor = 2
                trait = ambitious
            }
            health = -0.25
        }
    }
}

#each random change amounts to approx. 65% chance of gaining 1 point over three years, so we'd expect something like 2-3 stat points for students at elite institutions

        limit = {
            persistent_event_target:alma_mater = {

            chance = 30

        }

    }
}
