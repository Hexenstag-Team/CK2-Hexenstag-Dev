cults_grace_period_trigger = {
    OR  = {
        ai = yes
        total_years_played >= 5 #will cover ai = no in most cases, including the easier setting
        #now address edge cases in first five years
        has_game_rule = {
            name = warhammer_cult_grace_period
            value = hardmode
        }
        AND = {
            total_years_played >= 1
            has_game_rule = {
                name = warhammer_cult_grace_period
                value = default
            }
        }
    }
}

#used in event chains where the accused has already been identified as event_target:courtier_accused_by_witch_hunter
accused_courtier_is_secret_chaos_worshipper_trigger = {
    event_target:courtier_accused_by_witch_hunter = {
        OR = {
            true_religion_group = chaos_gods_group
            is_mutated_trigger = yes
        }
    }
}

    }
