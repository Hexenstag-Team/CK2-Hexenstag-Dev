
##CHANNELING###
offmap_decisions = {
	magic_channeling = {
		ai = no  #ai_will_do set to 0, so don't even check
		button_name = offmap_gifts
		ai_check_interval = 60

		potential = {
			is_offmap_tag = offmap_magic
		}

		from_potential = {
			magic_offmap_enabled_trigger = yes
		}

		allow = {
			FROM = {
				show_scope_change = no
				prisoner = no
				NOT = { trait = incapable }
				is_inaccessible_trigger = no
				NOT = { has_character_modifier = channeling }
			}
		}
		effect = {
			FROM = {
				add_character_modifier = {
					name = channeling
					years = 1
				}
				remove_character_modifier = casted_spell
				custom_tooltip = {
					text = spell_casting_tt
					mutation_toughness_test_effect = yes
				}
			}
		}

		revoke_allowed = {
			always = no
		}

		ai_will_do = {
			factor = 0
		}
	}
}

		button_name = offmap_boons
		ai_check_interval = 60

		potential = {
			is_offmap_tag = offmap_magic
		}

		from_potential = {
			trait = shadows_elemental
			has_magic_power_3 = yes
		}

		allow = {
			FROM = {
				show_scope_change = no
				prisoner = no
				NOT = { trait = incapable }
				is_inaccessible_trigger = no
				custom_tooltip = {
					text = casted_spell_tt
					NOT = { has_character_modifier = casted_spell }
				}
			}
		}
		effect = {
			FROM = {
				add_character_modifier = {
					modifier = dread_aspect_modifier
					duration = 90
				}
				spell_casting_3 = yes
			}
		}

		revoke_allowed = {
			always = no
		}

		ai_will_do = {
			factor = 1
			modifier = {
				factor = 0.25 # interval also increased - fewer checks with higher chance of success
			}
		}
	}
