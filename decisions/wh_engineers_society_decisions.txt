
targeted_decisions = {
  #decision to gamble on character health to gain society currency
	engineer_experiment = {
		is_in_society = yes
		filter = self
		ai_target_filter = self

		is_high_prio = yes

		potential = {
			character = FROM
			has_dlc = "Mystics"
			society_member_of = engineers_guild
			NOR = {
				has_character_modifier = impractical_innovation
				has_character_modifier = practical_innovation
				has_character_modifier = busy_engineer
			}
			is_incapable = no
			NOT = { is_inaccessible_trigger = yes }
		}

		allow = {
			custom_tooltip = {
				text = tooltip_not_busy
				NOT = { has_character_flag = do_not_disturb }
			}
		}

		effect = {
			custom_tooltip = {
				text = engineer_experiment_tt
				add_character_modifier = { name = busy_engineer duration = 30 }
				random_list = {
					60 = { #plans complete
            #knowledge base
						modifier = {
							factor = 2
							learning >= 15
						}
            #experience
						modifier = {
							factor = 2
							OR = {
								trait = dwarf_engineer
								trait = dwarf_engineer_master
								trait = engineer
							}
						}
            #"stickitude"
						modifier = {
							factor = 1.2
							trait = stubborn
						}
						character_event = {
							id = wh_engineers_society.10
							days = 14
							random = 7
						}
					}
					60 = { #can't make it work
            #lack of drive
						modifier = {
							factor = 1.5
							trait = slothful
						}
            #knowledge base
						modifier = {
							factor = 0.5
							learning >= 20
						}
            #drive
						modifier = {
							factor = 0.5
							trait = ambitious
						}
            #masterwork effect
            modifier = {
              factor = 0.5
              society_rank == 4
            }
						character_event = {
							id = wh_engineers_society.12
							days = 14
							random = 7
						}
					}
				}
			}
		}
		ai_will_do = {
			factor = 1
			modifier = {
				factor = 2
				OR = {
					trait = ambitious
					trait = proud
				}
			}
			modifier = {
				factor = 2
				OR = {
					trait = genius
					trait = brilliant
				}
			}
			modifier = {
				factor = 0.5
				learning < 12
			}
			modifier = {
				factor = 0.5
				trait = content
			}
      modifier = {
        factor = 0.1
        has_injury_trigger = yes
      }
		}
	}

}
