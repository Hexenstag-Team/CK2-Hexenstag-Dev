society_decisions = {
	shallyan_prosperity = {
		ai_check_interval = 36

		potential = {
			has_dlc = "Mystics"
			is_playable = yes
			society_rank = {
				rank >= 3
				society = cult_of_shallya
			}
			is_incapable = no
			prisoner = no
			NOT = {
				has_character_modifier = shallyan_prosperity_modifier
			}
		}

		allow = {
			has_society_currency_major_trigger = yes
			scaled_wealth = 0.25
		}

		effect = {
			if = {
				limit = {
					society_rank == 4
				}
				reduce_doom_progress_medium_effect = yes
				add_character_modifier = {
					name = shallyan_prosperity_modifier
					years = 10
				}
			}
			else = {
				reduce_doom_progress_tiny_effect = yes
				add_character_modifier = {
					name = shallyan_prosperity_modifier
					years = 5
				}
			}
			detract_society_currency_major_effect = yes
			scaled_wealth = -0.25
		}

		ai_will_do = {
			factor = 1
			modifier = {
				factor = 0.5
				trait = ambitious
			}
			modifier = {
				factor = 2
				trait = kind
			}
		}
	}
}
